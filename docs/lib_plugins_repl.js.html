topScope.exit = () => {
  console.log('\nPlease come back soon!'.blue);
};

topScope.help = () => {
  console.log('help()'.blue + ' shows this message'.green);
  console.log('exit() or CTRL-D'.blue + ' exits the REPL'.green);
};
